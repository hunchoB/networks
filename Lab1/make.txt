For S:
    S1:
    ! to open config print
    configure terminal
    ! to change hostname print
    hostname yourname
    ! to open config print
    configure terminal
    ! to set password
    enable secret switch1
    ! to show the resulting configuration
    show running-config
    ! to Save the Configuration
    copy running-config startup-config

    SSH:
    ! hostname and domain name REQUIRE
    ! set domain name
    # ip domain-name lab1.com
    ! generate key
    # crypto key generate rsa 
    ! size of key
    # 2048
    ! set the SSH version 2
    # ip ssh version 2
    ! configure vty lines
    # line vty 0 4
    # login local
    # exit
        !! disable transport protocols
        #transport input ssh
    ! define credentials
    # username hunchob secret switch1
